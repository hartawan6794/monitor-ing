CREATE TABLE `salesdetail` (
  `transid` bigint(20) NOT NULL AUTO_INCREMENT,
  `transdate` datetime DEFAULT NULL,
  `salesid` char(30) NOT NULL DEFAULT '',
  `salesidref` char(30) NOT NULL DEFAULT '',
  `productid` char(30) NOT NULL DEFAULT '',
  `snproduct` varchar(100) NOT NULL DEFAULT '',
  `expireddate` date DEFAULT NULL,
  `salesqty` double NOT NULL DEFAULT '0',
  `returnqty` double NOT NULL DEFAULT '0',
  `unit` char(10) NOT NULL DEFAULT '',
  `price` double NOT NULL DEFAULT '0',
  `grossamount` double NOT NULL DEFAULT '0',
  `taxid` tinyint(4) NOT NULL DEFAULT '0',
  `salestax` double NOT NULL DEFAULT '0',
  `returntax` double NOT NULL DEFAULT '0',
  `percentdisc` varchar(50) NOT NULL DEFAULT '',
  `valuedisc` double NOT NULL DEFAULT '0',
  `netamount` double NOT NULL DEFAULT '0',
  `cogs` double NOT NULL DEFAULT '0',
  `memo` text,
  `departement` char(10) NOT NULL DEFAULT '',
  `servicedoerid` char(30) NOT NULL DEFAULT '',
  `usercreate` char(16) NOT NULL DEFAULT '',
  `useredit` char(16) NOT NULL DEFAULT '',
  `updatetimestamp` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`transid`),
  KEY `salesid` (`salesid`),
  KEY `productid` (`productid`),
  KEY `unit` (`unit`),
  KEY `taxid` (`taxid`),
  KEY `departement` (`departement`),
  KEY `salesidref` (`salesidref`),
  KEY `servicedoerid` (`servicedoerid`),
  CONSTRAINT `salesdetail_ibfk_1` FOREIGN KEY (`salesid`) REFERENCES `sales` (`salesid`) ON UPDATE CASCADE,
  CONSTRAINT `salesdetail_ibfk_2` FOREIGN KEY (`productid`) REFERENCES `product` (`id`) ON UPDATE CASCADE,
  CONSTRAINT `salesdetail_ibfk_3` FOREIGN KEY (`departement`) REFERENCES `departement` (`id`) ON UPDATE CASCADE,
  CONSTRAINT `salesdetail_ibfk_4` FOREIGN KEY (`unit`) REFERENCES `units` (`unit`) ON UPDATE CASCADE,
  CONSTRAINT `salesdetail_ibfk_5` FOREIGN KEY (`taxid`) REFERENCES `taxes` (`id`) ON UPDATE CASCADE,
  CONSTRAINT `salesdetail_ibfk_6` FOREIGN KEY (`salesidref`) REFERENCES `sales` (`salesid`) ON UPDATE CASCADE,
  CONSTRAINT `salesdetail_ibfk_7` FOREIGN KEY (`servicedoerid`) REFERENCES `servicedoer` (`id`) ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=55809 DEFAULT CHARSET=utf8;
